const TYPE_META = {
  "Cert":     { color: "#FFB800", bg: "#FFB80018", label: "CERT"    },
  "Study":    { color: "#00C8FF", bg: "#00C8FF18", label: "STUDY"   },
  "Hands-on": { color: "#00FFB2", bg: "#00FFB218", label: "HANDS-ON"},
  "Project":  { color: "#C678FF", bg: "#C678FF18", label: "PROJECT" },
};

const YEAR_COLORS = { 1: "#00FFB2", 2: "#FF6B35" };

export default function Planner() {
  const [completed, setCompleted] = useState({});
  const [activeMonth, setActiveMonth] = useState(1);
  const [expanded, setExpanded] = useState(null);
  const [activeYear, setActiveYear] = useState(1);

  useEffect(() => {
    (async () => {
      try {
        const r = await window.storage.get("jec-24mo-progress");
        if (r) setCompleted(JSON.parse(r.value));
      } catch {}
    })();
  }, []);

  const save = async (next) => {
    try { await window.storage.set("jec-24mo-progress", JSON.stringify(next)); } catch {}
  };

  const toggle = (id) => {
    const next = { ...completed, [id]: !completed[id] };
    setCompleted(next);
    save(next);
  };

  const allTopics = MONTHS.flatMap(m => m.selfStudy);
  const totalDone = allTopics.filter(t => completed[t.id]).length;
  const totalPct  = Math.round((totalDone / allTopics.length) * 100);

  const yearMonths = MONTHS.filter(m => m.year === activeYear);
  const currentMonth = MONTHS.find(m => m.id === activeMonth) || MONTHS[0];
  const monthDone = currentMonth.selfStudy.filter(t => completed[t.id]).length;
  const monthPct  = Math.round((monthDone / currentMonth.selfStudy.length) * 100);

  const yc = YEAR_COLORS[currentMonth.year];

  // cert milestones for sidebar
  const certs = MONTHS.flatMap(m =>
    m.selfStudy.filter(t => t.type === "Cert").map(t => ({
      ...t, month: m.month, monthId: m.id
    }))
  );

  return (
    <div style={{
      minHeight:"100vh", background:"#080B10",
      fontFamily:"'IBM Plex Mono', 'Courier New', monospace",
      color:"#CBD5E0", display:"flex", flexDirection:"column",
    }}>
      {/* Top bar */}
      <div style={{
        background:"#0D1117", borderBottom:"1px solid #161B22",
        padding:"20px 20px 0", flexShrink:0,
      }}>
        <div style={{ display:"flex", justifyContent:"space-between", alignItems:"flex-start", flexWrap:"wrap", gap:12, marginBottom:16 }}>
          <div>
            <div style={{ fontSize:9, letterSpacing:5, color:"#30363D", marginBottom:4 }}>
              日本電子専門学校 · ネットワークセキュリティ科
            </div>
            <div style={{ fontSize:"clamp(20px,5vw,32px)", fontWeight:900, color:"#F0F6FC", letterSpacing:-1, lineHeight:1 }}>
              2-YEAR SELF-STUDY<br/>
              <span style={{ color: yc }}>MISSION PLAN</span>
            </div>
            <div style={{ fontSize:10, color:"#30363D", marginTop:5 }}>
              18 study months · 72 tasks · parallel curriculum
            </div>
          </div>
          {/* Global ring */}
          <div style={{ textAlign:"center" }}>
            <svg width="80" height="80" viewBox="0 0 80 80">
              <circle cx="40" cy="40" r="34" fill="none" stroke="#161B22" strokeWidth="7"/>
              <circle cx="40" cy="40" r="34" fill="none" stroke={yc} strokeWidth="7"
                strokeDasharray={`${2*Math.PI*34*totalPct/100} ${2*Math.PI*34}`}
                strokeLinecap="round"
                transform="rotate(-90 40 40)"
                style={{ transition:"stroke-dasharray 0.6s ease", filter:`drop-shadow(0 0 6px ${yc}80)` }}
              />
              <text x="40" y="44" textAnchor="middle" fill="#F0F6FC" fontSize="16" fontWeight="900"
                fontFamily="'IBM Plex Mono', monospace">{totalPct}%</text>
            </svg>
            <div style={{ fontSize:9, color:"#30363D", marginTop:2 }}>{totalDone}/{allTopics.length} tasks</div>
          </div>
        </div>

        {/* Year tabs */}
        <div style={{ display:"flex", gap:0, marginBottom:0 }}>
          {[1,2].map(y => (
            <button key={y} onClick={() => { setActiveYear(y); setActiveMonth(y===1?1:10); }}
              style={{
                padding:"10px 24px", background:"none", border:"none",
                borderBottom: activeYear===y ? `2px solid ${YEAR_COLORS[y]}` : "2px solid transparent",
                cursor:"pointer", color: activeYear===y ? YEAR_COLORS[y] : "#30363D",
                fontSize:11, fontFamily:"inherit", fontWeight: activeYear===y?700:400,
                letterSpacing:2, transition:"all 0.2s",
              }}>
              YEAR {y}
              <span style={{ marginLeft:8, fontSize:9, color: activeYear===y ? YEAR_COLORS[y]+"80" : "#30363D" }}>
                {y===1?"Month 1–9":"Month 10–18"}
              </span>
            </button>
          ))}
        </div>

        {/* Month strip */}
        <div style={{ display:"flex", gap:6, overflowX:"auto", padding:"12px 0 14px" }}>
          {yearMonths.map(m => {
            const done = m.selfStudy.filter(t => completed[t.id]).length;
            const all  = m.selfStudy.length;
            const full = done === all;
            const active = m.id === activeMonth;
            return (
              <button key={m.id} onClick={() => setActiveMonth(m.id)}
                style={{
                  flexShrink:0, padding:"8px 12px", cursor:"pointer",
                  background: active ? `${yc}15` : "transparent",
                  border: `1px solid ${active ? yc+"60" : full ? "#238636" : "#161B22"}`,
                  borderRadius:6, color: active ? yc : full ? "#238636" : "#30363D",
                  fontSize:10, fontFamily:"inherit", fontWeight: active?700:400,
                  letterSpacing:1, transition:"all 0.2s",
                  position:"relative",
                }}>
                {m.month.replace("Month ","")}
                {full && <span style={{ position:"absolute", top:-4, right:-4, fontSize:8, color:"#238636" }}>✓</span>}
              </button>
            );
          })}
        </div>
      </div>

      {/* Main content */}
      <div style={{ display:"flex", flex:1, minHeight:0, flexWrap:"wrap" }}>
        {/* Left: topic list */}
        <div style={{ flex:"1 1 320px", padding:"20px", minWidth:0, borderRight:"1px solid #161B22" }}>
          {/* Month header */}
          <div style={{ marginBottom:16 }}>
            <div style={{ display:"flex", alignItems:"center", gap:10, flexWrap:"wrap" }}>
              <span style={{
                fontSize:22, fontWeight:900, color:"#F0F6FC"
              }}>{currentMonth.month}</span>
              <span style={{
                fontSize:9, padding:"3px 10px", borderRadius:3,
                background:`${yc}15`, color:yc, border:`1px solid ${yc}40`, letterSpacing:2,
              }}>YEAR {currentMonth.year}</span>
            </div>
            <div style={{ fontSize:10, color:"#30363D", marginTop:4 }}>
              🏫 <span style={{ color:"#484F58" }}>{currentMonth.jecClass}</span>
            </div>
            {/* month progress */}
            <div style={{ marginTop:10, display:"flex", alignItems:"center", gap:10 }}>
              <div style={{ flex:1, height:3, background:"#161B22", borderRadius:2 }}>
                <div style={{
                  height:"100%", width:`${monthPct}%`, background:yc, borderRadius:2,
                  boxShadow:`0 0 6px ${yc}60`, transition:"width 0.4s ease",
                }}/>
              </div>
              <span style={{ fontSize:10, color: yc, fontWeight:700, minWidth:36 }}>{monthPct}%</span>
            </div>
          </div>

          {/* Tasks */}
          <div style={{ display:"flex", flexDirection:"column", gap:10 }}>
            {currentMonth.selfStudy.map(task => {
              const done = !!completed[task.id];
              const open = expanded === task.id;
              const tm   = TYPE_META[task.type];
              return (
                <div key={task.id} style={{
                  background: done ? "#0D1F0D" : "#0D1117",
                  border:`1px solid ${done?"#238636":"#161B22"}`,
                  borderRadius:8, overflow:"hidden",
                  transition:"all 0.25s",
                }}>
                  <div style={{ padding:"14px 16px", display:"flex", alignItems:"center", gap:12, cursor:"pointer" }}
                    onClick={() => setExpanded(open ? null : task.id)}>
                    {/* checkbox */}
                    <div onClick={e=>{ e.stopPropagation(); toggle(task.id); }}
                      style={{
                        width:20, height:20, borderRadius:5, flexShrink:0, cursor:"pointer",
                        border:`2px solid ${done?"#238636":"#30363D"}`,
                        background: done?"#238636":"transparent",
                        display:"flex", alignItems:"center", justifyContent:"center",
                        transition:"all 0.2s",
                      }}>
                      {done && <span style={{ fontSize:11, color:"#fff", fontWeight:900 }}>✓</span>}
                    </div>
                    {/* content */}
                    <div style={{ flex:1, minWidth:0 }}>
                      <div style={{ display:"flex", alignItems:"center", gap:8, flexWrap:"wrap" }}>
                        <span style={{ fontSize:13 }}>{task.icon}</span>
                        <span style={{
                          fontSize:13, fontWeight:700,
                          color: done?"#30363D":"#F0F6FC",
                          textDecoration: done?"line-through":"none",
                        }}>{task.title}</span>
                      </div>
                      <div style={{ marginTop:4, display:"flex", gap:6, flexWrap:"wrap", alignItems:"center" }}>
                        <span style={{
                          fontSize:9, padding:"2px 7px", borderRadius:3, letterSpacing:1,
                          background:tm.bg, color:tm.color, border:`1px solid ${tm.color}30`,
                        }}>{tm.label}</span>
                        <span style={{ fontSize:9, color:"#30363D" }}>~{task.hours}h/wk</span>
                      </div>
                    </div>
                    <span style={{
                      fontSize:10, color:"#30363D", flexShrink:0,
                      transform: open?"rotate(90deg)":"none", transition:"transform 0.2s",
                    }}>▶</span>
                  </div>

                  {open && (
                    <div style={{ padding:"0 16px 16px 48px", borderTop:"1px solid #161B22", paddingTop:12 }}>
                      <div style={{
                        background:"#080B10", borderRadius:6, padding:"10px 12px", marginBottom:10,
                        borderLeft:`3px solid ${yc}`,
                      }}>
                        <div style={{ fontSize:9, color:"#30363D", letterSpacing:2, marginBottom:4 }}>💡 TIP</div>
                        <div style={{ fontSize:11, color:"#8B949E", lineHeight:1.7 }}>{task.tip}</div>
                      </div>
                      <div style={{ fontSize:10, color:"#484F58", marginBottom:8 }}>
                        📚 {task.resource}
                      </div>
                      <a href={task.link} target="_blank" rel="noreferrer"
                        onClick={e=>e.stopPropagation()}
                        style={{
                          display:"inline-flex", alignItems:"center", gap:6, fontSize:10,
                          color:yc, textDecoration:"none", padding:"5px 12px",
                          borderRadius:5, border:`1px solid ${yc}40`, background:`${yc}10`,
                        }}>
                        → Open Resource
                      </a>
                    </div>
                  )}
                </div>
              );
            })}
          </div>

          {/* Month nav */}
          <div style={{ display:"flex", justifyContent:"space-between", marginTop:20 }}>
            <button onClick={() => {
                const prev = activeMonth - 1;
                if(prev >= 1){ setActiveMonth(prev); if(MONTHS.find(m=>m.id===prev)?.year !== currentMonth.year) setActiveYear(currentMonth.year===2?1:2); }
              }}
              disabled={activeMonth===1}
              style={{ padding:"8px 16px", background:"none", border:"1px solid #161B22", borderRadius:6,
                color:activeMonth===1?"#161B22":"#484F58", cursor:activeMonth===1?"default":"pointer",
                fontFamily:"inherit", fontSize:10 }}>
              ← Prev month
            </button>
            <button onClick={() => {
                const next = activeMonth + 1;
                if(next <= 18){ setActiveMonth(next); if(MONTHS.find(m=>m.id===next)?.year !== currentMonth.year) setActiveYear(currentMonth.year===1?2:1); }
              }}
              disabled={activeMonth===18}
              style={{ padding:"8px 16px", background: activeMonth===18?"none":`${yc}15`,
                border:`1px solid ${activeMonth===18?"#161B22":yc+"40"}`, borderRadius:6,
                color:activeMonth===18?"#161B22":yc, cursor:activeMonth===18?"default":"pointer",
                fontFamily:"inherit", fontSize:10, fontWeight:700 }}>
              Next month →
            </button>
          </div>
        </div>

        {/* Right: cert roadmap sidebar */}
        <div style={{ width:"220px", flexShrink:0, padding:"20px 16px", background:"#0D1117" }}>
          <div style={{ fontSize:9, color:"#30363D", letterSpacing:3, marginBottom:14 }}>CERT ROADMAP</div>
          <div style={{ display:"flex", flexDirection:"column", gap:0 }}>
            {certs.map((c, i) => {
              const done = !!completed[c.id];
              const active = c.monthId === activeMonth;
              return (
                <div key={c.id} style={{ display:"flex", gap:10, cursor:"pointer" }}
                  onClick={() => { setActiveMonth(c.monthId); setActiveYear(MONTHS.find(m=>m.id===c.monthId)?.year||1); }}>
                  {/* timeline */}
                  <div style={{ display:"flex", flexDirection:"column", alignItems:"center", width:18 }}>
                    <div style={{
                      width:14, height:14, borderRadius:"50%", flexShrink:0,
                      background: done?"#238636": active?`${YEAR_COLORS[MONTHS.find(m=>m.id===c.monthId)?.year||1]}`:"#161B22",
                      border:`2px solid ${done?"#238636": active?YEAR_COLORS[MONTHS.find(m=>m.id===c.monthId)?.year||1]:"#30363D"}`,
                      display:"flex", alignItems:"center", justifyContent:"center",
                      boxShadow: active ? `0 0 8px ${YEAR_COLORS[MONTHS.find(m=>m.id===c.monthId)?.year||1]}80` : "none",
                      transition:"all 0.3s",
                    }}>
                      {done && <span style={{ fontSize:7, color:"#fff" }}>✓</span>}
                    </div>
                    {i < certs.length-1 && <div style={{ width:1, flex:1, minHeight:16, background:"#161B22" }}/>}
                  </div>
                  {/* label */}
                  <div style={{ paddingBottom:14 }}>
                    <div style={{
                      fontSize:10, fontWeight: active?700:400,
                      color: done?"#238636": active?YEAR_COLORS[MONTHS.find(m=>m.id===c.monthId)?.year||1]:"#484F58",
                      lineHeight:1.3, marginBottom:2,
                    }}>{c.title.replace(" ← TAKE IT 🎉","").replace(" ← TAKE IT 🏆","").replace(" – prep start","").replace(" – final prep (exam in Month 9)","")}</div>
                    <div style={{ fontSize:8, color:"#30363D" }}>{c.month}</div>
                  </div>
                </div>
              );
            })}
          </div>

          {/* Legend */}
          <div style={{ marginTop:20, borderTop:"1px solid #161B22", paddingTop:14 }}>
            <div style={{ fontSize:9, color:"#30363D", letterSpacing:3, marginBottom:10 }}>TASK TYPES</div>
            {Object.entries(TYPE_META).map(([k,v]) => (
              <div key={k} style={{ display:"flex", alignItems:"center", gap:8, marginBottom:6 }}>
                <div style={{ width:6, height:6, borderRadius:"50%", background:v.color, flexShrink:0 }}/>
                <span style={{ fontSize:9, color:"#484F58" }}>{k}</span>
              </div>
            ))}
          </div>

          <div style={{ marginTop:16, fontSize:9, color:"#30363D", lineHeight:1.6 }}>
            Progress saved<br/>automatically ✓
          </div>
        </div>
      </div>
    </div>
  );
}
